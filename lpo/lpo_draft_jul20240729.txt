# Real-World Place Class
lpo:RealWorldPlace a rdfs:Class ;
    rdfs:label "Real-World Place" ;
    rdfs:comment "An actual place in the world." .

# Place Description Class
lpo:Place a rdfs:Class ;
    rdfs:label "Place" ;
    rdfs:comment "A description of a place in the world." .

# Describes relationship
lpo:describes a rdfs:ObjectProperty ;
    rdfs:domain lpo:Place ;
    rdfs:range lpo:RealWorldPlace ;
    rdfs:label "describes" ;
    rdfs:comment "Relates a place description to a real-world place." .
